var blub_sort(arr)
{
	var n = count(arr);
	var i = 1;
	
	while(i < n)
	{
		var k = i;
		while(k > 0)
		{
			if(arr[k] < arr[k-1])
			{
				var tmp = arr[k];
				arr[k] = arr[k-1];
				arr[k-1] = tmp;
				k--;
			}else
			{
				break;
			}
			
		}
		i++;
	}
}
